alias War_TongKet {

  var %point.bot = %point. [ $+ [ $me ] ]
  var %post.bot = %post. [ $+ [ $me ] ]
  :tongketwar
  if (%line <= $lines(hethong\CasinoWar.txt)) {
    var %item = $read(hethong\CasinoWar.txt, %line)
    var %nick = $uw(%item,1,=)

    var %post.player = %post. [ $+ [ %nick ] ]

        ghidiem diem %nick $int($calc($coidiem(diem,%nick) + %bonus2))
